			if = {
				limit = {
					RAJ = {
						check_variable = {
							var = ENG_RAJ_revolt_risk
							value = 2
							compare = greater_than_or_equals
						}
					}
				}
				RAJ = {
					subtract_from_variable = {
						var = ENG_RAJ_revolt_risk
						value = 2
					}
				}
			}
			else_if = {
				limit = {
					RAJ = {
						check_variable = {
							var = ENG_RAJ_revolt_risk
							value = 1
							compare = greater_than_or_equals
						}
					}
				}
				RAJ = {
					subtract_from_variable = {
						var = ENG_RAJ_revolt_risk
						value = 1
					}
				}
			}
			else = {
				RAJ = { add_stability = 0.05 }
			}
		}
	}
	
}
